# 隐藏参数挤出空行问题完全指南

## 问题描述

在 ComfyUI 节点中，当使用 `hidden: true` 隐藏某些控件参数时，会出现以下问题：
1. 隐藏的参数虽然不可见，但会在面板中留下空白占位
2. 节点高度不正确，出现多余的空白区域
3. 参数越多，空白区域越大

## 根本原因

ComfyUI 的 `hidden: true` 只是将参数标记为隐藏，但不会：
- 将控件的尺寸设置为 0
- 调整节点的整体高度计算
- 隐藏相关的 DOM 元素

## 最佳解决方案

### 1. 使用 GJJ_Utils.hideWidget()

GJJ 公共工具类 `gjj_utils.js` 提供了标准的隐藏方法：

```javascript
import { GJJ_Utils } from "./gjj_utils.js";

// 隐藏控件
GJJ_Utils.hideWidget(widget);

// 刷新节点尺寸
GJJ_Utils.refreshNode(node);
```

**hideWidget 方法会执行以下操作：**
- 设置 `widget.hidden = true`
- 将 `widget.type` 改为 `converted-widget:NAME`
- 设置 `widget.computeSize = () => [0, -4]`
- 设置 `widget.getHeight = () => -4`
- 将 `widget.y` 和 `widget.last_y` 设置为 -10000
- 设置 DOM 元素 `display: none`

### 2. 多层防护机制

隐藏操作需要在多个生命周期钩子中执行：

```javascript
function setupNode(node) {
    // 方法1: 在 onNodeCreated 中隐藏
    const origOnNodeCreated = nodeType.prototype.onNodeCreated;
    nodeType.prototype.onNodeCreated = function() {
        const result = origOnNodeCreated?.call(this);
        requestAnimationFrame(() => {
            // 隐藏参数
            hiddenParams.forEach(name => {
                const w = GJJ_Utils.getWidget(this, name);
                if (w) GJJ_Utils.hideWidget(w);
            });
            GJJ_Utils.refreshNode(this);
        });
        return result;
    };

    // 方法2: 在 onConfigure 中隐藏
    const origOnConfigure = node.onConfigure;
    node.onConfigure = function(...args) {
        if (origOnConfigure) origOnConfigure.apply(this, args);
        requestAnimationFrame(() => {
            updateVisibility(this);
        });
    };

    // 方法3: 在 nodeCreated 中隐藏
    app.registerExtension({
        nodeCreated(node) {
            requestAnimationFrame(() => {
                setupNode(node);
            });
        }
    });
}
```

### 3. 动态显示/隐藏参数

```javascript
function updateVisibility(node) {
    const showSDParams = node.widgets.find(w => w.name === "mode")?.value === "sd_mode";

    const sdParams = ["cfg", "scheduler", "steps"];
    const fluxParams = ["guidance", "clip_model"];

    sdParams.forEach(name => {
        const w = GJJ_Utils.getWidget(node, name);
        if (w) {
            if (showSDParams) {
                showWidget(w);  // 显示
            } else {
                GJJ_Utils.hideWidget(w);  // 隐藏
            }
        }
    });

    GJJ_Utils.refreshNode(node);
}

function showWidget(widget) {
    if (!widget) return;
    widget.__gjjUtilsHidden = false;
    widget.hidden = false;
    if (widget.element) {
        widget.element.style.display = "";
        widget.element.style.visibility = "";
    }
    if (widget.inputEl) {
        widget.inputEl.style.display = "";
        widget.inputEl.style.visibility = "";
    }
    widget.computeSize = undefined;
    widget.getHeight = undefined;
    widget.draw = undefined;
    widget.y = undefined;
    widget.last_y = undefined;
}
```

## 关键要点总结

### ✅ 应该做的

1. **使用完整的隐藏函数**：不要只设置 `hidden = true`
2. **重点检查 `last_y`**：这个属性最关键，必须设置为负值
3. **多层防护**：在 `addWidget`、`onNodeCreated`、`onConfigure` 三个时机都进行隐藏
4. **刷新节点**：隐藏后必须调用 `GJJ_Utils.refreshNode(node)`
5. **阻止事件冒泡**：所有 DOM 事件都要调用 `event.stopPropagation()`

### ❌ 不应该做的

1. **不要只设置 `hidden = true`**：这远远不够
2. **不要忽略 `last_y`**：这是最常见的遗漏
3. **不要只在单一时机隐藏**：需要在多个生命周期钩子中都进行处理
4. **不要忘记刷新节点**：会导致空白区域
5. **不要使用 `cloneNode`**：会丢失事件监听器

## 完整示例

```javascript
import { app } from "/scripts/app.js";
import { GJJ_Utils } from "./gjj_utils.js";

const NODE_TYPE = "GJJ_CustomNode";

const PARAMS_BY_MODE = {
    mode_a: ["param1", "param2", "param3"],
    mode_b: ["param4", "param5", "param6"],
};

function updateParameterVisibility(node, mode) {
    const allParams = Object.values(PARAMS_BY_MODE).flat();
    const visibleParams = PARAMS_BY_MODE[mode] || [];

    allParams.forEach(name => {
        const w = GJJ_Utils.getWidget(node, name);
        if (!w) return;

        if (visibleParams.includes(name)) {
            showWidget(w);
        } else {
            GJJ_Utils.hideWidget(w);
        }
    });

    GJJ_Utils.refreshNode(node);
}

function showWidget(widget) {
    if (!widget) return;
    widget.__gjjUtilsHidden = false;
    widget.hidden = false;
    if (widget.element) widget.element.style.display = "";
    if (widget.inputEl) widget.element.style.display = "";
    widget.computeSize = undefined;
    widget.getHeight = undefined;
    widget.y = undefined;
    widget.last_y = undefined;
}

function setupNode(node) {
    if (node._setupComplete) return;
    node._setupComplete = true;

    const modeWidget = GJJ_Utils.getWidget(node, "mode");
    if (modeWidget) {
        modeWidget.callback = (value) => {
            updateParameterVisibility(node, value);
        };
    }

    requestAnimationFrame(() => {
        const mode = modeWidget?.value || "mode_a";
        updateParameterVisibility(node, mode);
    });
}

app.registerExtension({
    name: "GJJ.CustomNode",

    beforeRegisterNodeDef(nodeType, nodeData, app) {
        if (nodeData.name !== NODE_TYPE) return;

        const origOnNodeCreated = nodeType.prototype.onNodeCreated;
        nodeType.prototype.onNodeCreated = function() {
            const result = origOnNodeCreated?.call(this);
            requestAnimationFrame(() => setupNode(this));
            return result;
        };
    },

    nodeCreated(node) {
        if (node.type !== NODE_TYPE) return;
        requestAnimationFrame(() => setupNode(node));
    },
});
```

## 常见问题排查

1. **隐藏后仍有空白**：检查 `last_y` 是否正确设置
2. **节点高度不正确**：调用 `GJJ_Utils.refreshNode(node)`
3. **参数不可见**：确保调用了 `showWidget` 函数
4. **切换模式后状态不变**：检查 callback 函数是否正确设置
